  <script>
    (function () {
      const tabs = document.querySelectorAll(".tab");
      const panels = document.querySelectorAll(".tab-panel");

      function setActiveTab(name) {
        tabs.forEach((t) => {
          t.classList.toggle("active", t.dataset.tab === name);
        });
        panels.forEach((p) => {
          p.style.display = p.dataset.panel === name ? "block" : "none";
        });
      }

      tabs.forEach((tab) => {
        tab.addEventListener("click", () => setActiveTab(tab.dataset.tab));
      });

      const statusFile = document.getElementById("statusFile");
      const statusUrl = document.getElementById("statusUrl");
      const statusXml = document.getElementById("statusXml");

      function setStatus(el, type, msg) {
        el.classList.remove("ok", "err");
        if (type === "ok") el.classList.add("ok");
        if (type === "err") el.classList.add("err");
        const span = el.querySelector("span");
        if (span) span.innerHTML = msg;
      }

      const dl = document.getElementById("hiddenDownload");
      function triggerDownloadFromBlob(blob, filename) {
        const url = URL.createObjectURL(blob);
        dl.href = url;
        dl.download = filename || "profile.mobileconfig";
        dl.click();
        setTimeout(() => URL.revokeObjectURL(url), 5000);
      }

      // FILE
      const fileInput = document.getElementById("fileInput");
      const btnInstallFile = document.getElementById("btnInstallFile");
      const btnClearFile = document.getElementById("btnClearFile");

      btnInstallFile.addEventListener("click", () => {
        const file = fileInput.files && fileInput.files[0];
        if (!file) {
          setStatus(statusFile, "err", "<strong>No file selected.</strong> Choose a .mobileconfig file first.");
          return;
        }
        const reader = new FileReader();
        btnInstallFile.disabled = true;
        setStatus(statusFile, "ok", "Reading file…");
        reader.onload = function (e) {
          try {
            const data = e.target.result;
            const blob = new Blob([data], {
              type: "application/x-apple-aspen-config",
            });
            const name = file.name && file.name.endsWith(".mobileconfig")
              ? file.name
              : (file.name || "profile") + ".mobileconfig";
            triggerDownloadFromBlob(blob, name);
            setStatus(
              statusFile,
              "ok",
              "<strong>Ready.</strong> Safari should prompt to install the profile."
            );
          } catch (err) {
            console.error(err);
            setStatus(
              statusFile,
              "err",
              "<strong>Error.</strong> Could not process file."
            );
          } finally {
            btnInstallFile.disabled = false;
          }
        };
        reader.onerror = function () {
          btnInstallFile.disabled = false;
          setStatus(
            statusFile,
            "err",
            "<strong>Error.</strong> Failed to read file."
          );
        };
        reader.readAsArrayBuffer(file);
      });

      btnClearFile.addEventListener("click", () => {
        fileInput.value = "";
        setStatus(statusFile, "", "Waiting for file…");
      });

      // URL
      const urlInput = document.getElementById("urlInput");
      const btnInstallUrl = document.getElementById("btnInstallUrl");
      const btnClearUrl = document.getElementById("btnClearUrl");

      btnInstallUrl.addEventListener("click", async () => {
        const url = (urlInput.value || "").trim();
        if (!url) {
          setStatus(statusUrl, "err", "<strong>No URL.</strong> Enter a direct .mobileconfig link.");
          return;
        }
        try {
          btnInstallUrl.disabled = true;
          setStatus(statusUrl, "ok", "Fetching profile…");
          const res = await fetch(url, { credentials: "omit" });
          if (!res.ok) {
            throw new Error("HTTP " + res.status);
          }
          const buf = await res.arrayBuffer();
          const blob = new Blob([buf], {
            type: "application/x-apple-aspen-config",
          });
          const filenameFromUrl = (() => {
            try {
              const u = new URL(url);
              const path = u.pathname.split("/").filter(Boolean).pop() || "";
              if (path.toLowerCase().endsWith(".mobileconfig")) return path;
              return "profile.mobileconfig";
            } catch {
              return "profile.mobileconfig";
            }
          })();
          triggerDownloadFromBlob(blob, filenameFromUrl);
          setStatus(
            statusUrl,
            "ok",
            "<strong>Ready.</strong> Safari should prompt to install the profile."
          );
        } catch (err) {
          console.error(err);
          setStatus(
            statusUrl,
            "err",
            "<strong>Error.</strong> Could not fetch profile. Check URL and CORS."
          );
        } finally {
          btnInstallUrl.disabled = false;
        }
      });

      btnClearUrl.addEventListener("click", () => {
        urlInput.value = "";
        setStatus(statusUrl, "", "Waiting for URL…");
      });

      // XML
      const xmlInput = document.getElementById("xmlInput");
      const btnInstallXml = document.getElementById("btnInstallXml");
      const btnClearXml = document.getElementById("btnClearXml");

      btnInstallXml.addEventListener("click", () => {
        const xml = xmlInput.value.trim();
        if (!xml) {
          setStatus(statusXml, "err", "<strong>No XML.</strong> Paste configuration XML first.");
          return;
        }
        btnInstallXml.disabled = true;
        try {
          const blob = new Blob([xml], {
            type: "application/x-apple-aspen-config",
          });
          triggerDownloadFromBlob(blob, "profile-from-xml.mobileconfig");
          setStatus(
            statusXml,
            "ok",
            "<strong>Ready.</strong> Safari should prompt to install the profile."
          );
        } catch (err) {
          console.error(err);
          setStatus(
            statusXml,
            "err",
            "<strong>Error.</strong> Could not generate profile from XML."
          );
        } finally {
          btnInstallXml.disabled = false;
        }
      });

      btnClearXml.addEventListener("click", () => {
        xmlInput.value = "";
        setStatus(statusXml, "", "Waiting for XML…");
      });

      // Small Safari hint if not Safari on iOS
      (function safariHint() {
        const ua = navigator.userAgent || "";
        const isIOS = /iPad|iPhone|iPod/.test(ua);
        const isSafari = /^((?!chrome|android).)*safari/i.test(ua);
        if (!isIOS || !isSafari) return;
        // Already in Safari on iOS – nothing extra needed.
      })();
    })();
  </script>
